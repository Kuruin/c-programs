        #include <stdio.h>
        #include<conio.h>
        void swap(int *n1, int *n2)
        {
            int temp=*n1;
            *n1=*n2;
            *n2=temp;
        }
        int main() {
            int n1,n2;
            printf("Enter two numbers: ");
            scanf("%d%d",&n1,&n2);
            printf("Vlaues before swapping :\n N1=%d N2=%d \n",n1,n2);
            swap(&n1,&n2);
            printf("Vlaues after swapping :\n N1=%d N2=%d",n1,n2);
            
            return 0;
        }

Q7:- 
        #include <stdio.h>
        int factorial(int n)
        {
            if(n>=1)
            {
                return n *factorial(n-1);
            }
            else {
                return 1;
            }
        }
        int main() {
            int n;
            printf("Enter a number: ");
            scanf("%d",&n);
            printf("The factorial of %d is %d",n,factorial(n));
            return 0;
        }

        #include <stdio.h>
        #include<string.h>
        int main() {
            int a[100],n,i,j,temp=0;
            printf("How many elements: ");
            scanf("%d",&n);
            printf("Enter the elements to be sorted: \n");
            for(i=0;i<n;i++)
            {
                scanf("%d",&a[i]);
                
            }
            printf("The sorted elements are: \n");
            for(i=0;i<n;i++)
            {
                for(j=i+1;j<n;j++)
                {
                    if(a[i]>a[j])
                    {
                        temp=a[i];
                        a[i]=a[j];
                        a[j]=temp;
                    }
                
            }
            }
            for(i=0;i<n;i++)
            {
                printf("%d \n",a[i]);
            }
            return 0;
        }

        #include <stdio.h>
        #include<string.h>
        int main() {
            char str[100];
            int len,i,flag=0;
            printf("Enter a string: ");
            gets(str);
            len=strlen(str);
            for(i=0;i<len;i++)
            {
                if(str[i]!=str[len-i-1])
                {
                    flag=1;
                    break;
                }
                
            }
            if(flag==0)
            {
                printf("%s is  a palindrome string",str);
            }
            else{
                printf("%s is  not a palindrome string",str);
            }
            return 0;
        }
